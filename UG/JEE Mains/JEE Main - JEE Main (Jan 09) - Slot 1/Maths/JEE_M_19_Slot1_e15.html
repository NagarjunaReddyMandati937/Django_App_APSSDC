\[ \vec {a} \times \vec {c} = \vec {-b} \]


    \[ (\vec {a} \times \vec {c}) \times \vec {a} = -\vec {b} \times \vec {a} \]


    \[ \Rightarrow (\vec {a} \times \vec {c})\times \vec {a} = \vec {a}\times \vec {b} \]


    \[ \Rightarrow (\vec {a} . \vec {a})\vec {c} - (\vec {c} . \vec {a}) \vec {a} = \vec {a}\times \vec {b} \]


    \[ \Rightarrow 2\vec {c} - 4\vec {a} = \vec {a}\times \vec {b} \]


    Now

    \[ \vec {a}\times \vec {b} = \begin{vmatrix}\hat {i} & \hat {j} & \hat {k}\\ 1 & -1 & 0\\ 1 & 1 &1 \end{vmatrix} = -\hat {i} - \hat {j} + 2\hat {k} \]


    So,

    \[ 2\vec {c} = 4\hat {i} - 4\hat {j} - \hat {i} - \hat {j} + 2\hat {k} \]


    \[ = 3\hat {i} - 5\hat {j} + 2\hat {k} \]


    \[ \Rightarrow \vec {c} = \dfrac {3}{2}\hat {i} - \dfrac {5}{2}\hat {j} + \hat {k} \]


    \[ |\vec {c}| = \sqrt {\dfrac {9}{4} + \dfrac {25}{4} + 1} \]


    \[ = \sqrt {\dfrac {38}{4}} = \sqrt {\dfrac {19}{2}} \]


    \[|\vec {c}|^{2} = \dfrac {19}{2}\]
