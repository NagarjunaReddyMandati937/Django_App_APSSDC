\[ \left( {{x^2} - {y^2}} \right)dx + 2xydy = 0 \]


    \[ \Rightarrow 2xydy = ({y^2} - {x^2})dx \]


    \[ \Rightarrow \dfrac{{dy}}{{dx}} = \dfrac{{{y^2} - {x^2}}}{{2xy}} \]


    putting

    \[ y = vx \]


    \[ \Rightarrow\dfrac{{dy}}{{dx}} = v + x\dfrac{{dv}}{{dx}} \]


    Now,

    \[ v + x\dfrac{{dv}}{{dx}} = \dfrac{{{v^2}{x^2} - {x^2}}}{{2x \times vx}} \]


    \[ \Rightarrow x\dfrac{{dv}}{{dx}} = \dfrac{{{v^2} - 1}}{{2v}} - v \]


    \[ \Rightarrow x\dfrac{{dv}}{{dx}} = - \dfrac{{\left( {{v^2} + 1} \right)}}{{2v}} \]


    \[ \Rightarrow \int_{}^{} {\dfrac{{ - 2v}}{{{v^2} + 1}}dv = \int_{}^{} {\dfrac{{dx}} {x}} } \]


    Putting

    \[ {v^2} + 1 = t \]


    \[ 2vdv = dt \]


    \[ \Rightarrow - \int_{}^{} {\dfrac{{dt}}{t} = \int_{}^{} {\dfrac{{dx}}{x}} } \]


    \[ \Rightarrow - {\mathop{\rm logt}\nolimits} = \log x + {c_1} \]


    \[ \Rightarrow \dfrac{1}{t} = x{c_1} \]


    \[ \Rightarrow \dfrac{1}{{{v^2} + 1}} = x{c_1} \]


    \[ \Rightarrow \dfrac{1}{{\dfrac{{{y^2}}}{{{x^2}}} + 1}} = x{c_1} \]


    \[ \Rightarrow \dfrac{{{x^2}}}{{{x^2} + {y^2}}} = x{c_1} \]


    \[ \Rightarrow {x^2} + {y^2} = \dfrac{x}{{{c_1}}} \]


    \[ \Rightarrow {x^2} + {y^2} = xc \]
